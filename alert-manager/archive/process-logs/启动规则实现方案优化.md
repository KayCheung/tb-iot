## 二、启动规则 TBEL 实现

### 2.1 ALWAYS（始终启动）

```javascript
function expression(ctx, temperature) {
  // 不添加时间判断，只判断指标条件
  return temperature > 80;
}
```

### 2.2 SCHEDULED（定时启动）

#### 配置示例
```json
{
  "enableRuleType": "SCHEDULED",
  "enableRuleConfig": {
    "startDate": "2026-03-01",
    "endDate": "2026-12-31",
    "startTime": "08:00",
    "endTime": "18:00"
  }
}
```

#### TBEL 实现
```javascript
function expression(ctx, temperature) {
  var currentTs = ctx.latestTs;
  var currentDate = new Date(currentTs);
  
  // 日期范围判断
  var startDate = new Date('2026-03-01').getTime();
  var endDate = new Date('2026-12-31').getTime();
  if (currentTs < startDate || currentTs > endDate) {
    return false;
  }
  
  // 时间段判断
  var hour = currentDate.getHours();
  var minute = currentDate.getMinutes();
  var currentMinutes = hour * 60 + minute;
  
  var startMinutes = 8 * 60 + 0;  // 08:00
  var endMinutes = 18 * 60 + 0;   // 18:00
  
  if (currentMinutes < startMinutes || currentMinutes >= endMinutes) {
    return false;
  }
  
  // 指标条件判断
  return temperature > 80;
}
```

### 2.3 CUSTOM（自定义时间）

#### 配置示例
```json
{
  "enableRuleType": "CUSTOM",
  "enableRuleConfig": {
    "weekDays": [1, 2, 3, 4, 5],  // 周一到周五
    "timeRanges": [
      {"start": "08:00", "end": "12:00"},
      {"start": "14:00", "end": "18:00"}
    ]
  }
}
```

#### TBEL 实现
```javascript
function expression(ctx, temperature) {
  var currentTs = ctx.latestTs;
  var currentDate = new Date(currentTs);
  
  // 星期判断（0=周日, 1=周一, ..., 6=周六）
  var day = currentDate.getDay();
  var allowedDays = [1, 2, 3, 4, 5]; // 周一到周五
  if (allowedDays.indexOf(day) === -1) {
    return false;
  }
  
  // 时间段判断
  var hour = currentDate.getHours();
  var minute = currentDate.getMinutes();
  var currentMinutes = hour * 60 + minute;
  
  // 检查是否在任一时间段内
  var inTimeRange = false;
  
  // 时间段1: 08:00-12:00
  if (currentMinutes >= 8 * 60 && currentMinutes < 12 * 60) {
    inTimeRange = true;
  }
  
  // 时间段2: 14:00-18:00
  if (currentMinutes >= 14 * 60 && currentMinutes < 18 * 60) {
    inTimeRange = true;
  }
  
  if (!inTimeRange) {
    return false;
  }
  
  // 指标条件判断
  return temperature > 80;
}
```

## 三、持续时间（Duration）实现

### 3.1 配置示例

```json
{
  "alarmLevel": "CRITICAL",
  "conditionOperator": "GT",
  "thresholdValue": 80,
  "durationSeconds": 300  // 持续 5 分钟
}
```

### 3.2 TBEL 实现

```javascript
function expression(ctx, temperature) {
  var currentTs = ctx.latestTs;
  var temperatureTs = ctx.args.temperature.ts;
  
  // 计算持续时间（秒）
  var durationSeconds = (currentTs - temperatureTs) / 1000;
  
  // 温度超过 80 度，且持续 300 秒
  return temperature > 80 && durationSeconds >= 300;
}
```

### 3.3 注意事项

⚠️ **持续时间的理解**：
- `ctx.args.temperature.ts` 是该指标最后一次更新的时间戳
- `currentTs - temperatureTs` 表示该值保持不变的时间
- 如果值频繁变化，持续时间会重置

**更准确的持续时间判断**：

如果需要判断"温度持续超过 80 度 5 分钟"，需要：
1. 记录首次超过 80 度的时间
2. 判断当前时间 - 首次超过时间 >= 300 秒

这种逻辑在 TBEL 中较难实现，建议：
- **简单场景**：使用 `ctx.args.xxx.ts`（值不变的时间）
- **复杂场景**：在告警系统侧实现（记录首次触发时间）

## 五、TBEL 脚本生成器优化

### 5.1 生成器接口

```java
public class TbelScriptGenerator {
    
    /**
     * 生成完整的 TBEL 脚本（包含启动规则）
     */
    public String generateTbelScript(
        AlarmScheme scheme,
        List<AlarmSchemeMetricRule> metricRules,
        Map<Long, List<AlarmLevelCondition>> levelConditionsMap
    ) {
        StringBuilder script = new StringBuilder();
        
        // 1. 函数签名
        script.append(generateFunctionSignature(metricRules));
        
        // 2. 变量定义
        script.append("  var currentTs = ctx.latestTs;\n");
        script.append("  var currentDate = new Date(currentTs);\n");
        
        // 3. 启动规则判断
        String enableRuleCheck = generateEnableRuleCheck(scheme);
        if (enableRuleCheck != null) {
            script.append(enableRuleCheck);
        }
        
        // 4. 指标条件判断
        String metricConditions = generateMetricConditions(metricRules, levelConditionsMap);
        script.append("  return ").append(metricConditions).append(";\n");
        
        script.append("}\n");
        
        return script.toString();
    }
    
    /**
     * 生成启动规则检查代码
     */
    private String generateEnableRuleCheck(AlarmScheme scheme) {
        String ruleType = scheme.getEnableRuleType();
        JSONObject config = scheme.getEnableRuleConfig();
        
        switch (ruleType) {
            case "ALWAYS":
                return null; // 不需要时间判断
                
            case "SCHEDULED":
                return generateScheduledCheck(config);
                
            case "CUSTOM":
                return generateCustomCheck(config);
                
            case "CONDITION":
                return generateConditionCheck(config);
                
            default:
                return null;
        }
    }
    
    /**
     * 生成 SCHEDULED 规则检查
     */
    private String generateScheduledCheck(JSONObject config) {
        String startDate = config.getString("startDate");
        String endDate = config.getString("endDate");
        String startTime = config.getString("startTime");
        String endTime = config.getString("endTime");
        
        StringBuilder code = new StringBuilder();
        
        // 日期范围判断
        code.append("  // 日期范围判断\n");
        code.append(String.format("  var startDate = new Date('%s').getTime();\n", startDate));
        code.append(String.format("  var endDate = new Date('%s').getTime();\n", endDate));
        code.append("  if (currentTs < startDate || currentTs > endDate) {\n");
        code.append("    return false;\n");
        code.append("  }\n\n");
        
        // 时间段判断
        String[] startParts = startTime.split(":");
        String[] endParts = endTime.split(":");
        int startMinutes = Integer.parseInt(startParts[0]) * 60 + Integer.parseInt(startParts[1]);
        int endMinutes = Integer.parseInt(endParts[0]) * 60 + Integer.parseInt(endParts[1]);
        
        code.append("  // 时间段判断\n");
        code.append("  var hour = currentDate.getHours();\n");
        code.append("  var minute = currentDate.getMinutes();\n");
        code.append("  var currentMinutes = hour * 60 + minute;\n");
        code.append(String.format("  if (currentMinutes < %d || currentMinutes >= %d) {\n", startMinutes, endMinutes));
        code.append("    return false;\n");
        code.append("  }\n\n");
        
        return code.toString();
    }
    
    /**
     * 生成 CUSTOM 规则检查
     */
    private String generateCustomCheck(JSONObject config) {
        JSONArray weekDays = config.getJSONArray("weekDays");
        JSONArray timeRanges = config.getJSONArray("timeRanges");
        
        StringBuilder code = new StringBuilder();
        
        // 星期判断
        code.append("  // 星期判断\n");
        code.append("  var day = currentDate.getDay();\n");
        code.append("  var allowedDays = ").append(weekDays.toString()).append(";\n");
        code.append("  if (allowedDays.indexOf(day) === -1) {\n");
        code.append("    return false;\n");
        code.append("  }\n\n");
        
        // 时间段判断
        code.append("  // 时间段判断\n");
        code.append("  var hour = currentDate.getHours();\n");
        code.append("  var minute = currentDate.getMinutes();\n");
        code.append("  var currentMinutes = hour * 60 + minute;\n");
        code.append("  var inTimeRange = false;\n\n");
        
        for (int i = 0; i < timeRanges.size(); i++) {
            JSONObject range = timeRanges.getJSONObject(i);
            String start = range.getString("start");
            String end = range.getString("end");
            
            String[] startParts = start.split(":");
            String[] endParts = end.split(":");
            int startMinutes = Integer.parseInt(startParts[0]) * 60 + Integer.parseInt(startParts[1]);
            int endMinutes = Integer.parseInt(endParts[0]) * 60 + Integer.parseInt(endParts[1]);
            
            code.append(String.format("  // 时间段%d: %s-%s\n", i + 1, start, end));
            code.append(String.format("  if (currentMinutes >= %d && currentMinutes < %d) {\n", startMinutes, endMinutes));
            code.append("    inTimeRange = true;\n");
            code.append("  }\n\n");
        }
        
        code.append("  if (!inTimeRange) {\n");
        code.append("    return false;\n");
        code.append("  }\n\n");
        
        return code.toString();
    }
    
    /**
     * 生成 CONDITION 规则检查
     */
    private String generateConditionCheck(JSONObject config) {
        String conditionMetric = config.getString("conditionMetric");
        String conditionOperator = config.getString("conditionOperator");
        String conditionValue = config.getString("conditionValue");
        
        StringBuilder code = new StringBuilder();
        code.append("  // 工况条件判断\n");
        
        switch (conditionOperator) {
            case "EQ":
                code.append(String.format("  if (%s !== '%s') {\n", conditionMetric, conditionValue));
                break;
            case "NEQ":
                code.append(String.format("  if (%s === '%s') {\n", conditionMetric, conditionValue));
                break;
            case "GT":
                code.append(String.format("  if (%s <= %s) {\n", conditionMetric, conditionValue));
                break;
            case "LT":
                code.append(String.format("  if (%s >= %s) {\n", conditionMetric, conditionValue));
                break;
        }
        
        code.append("    return false;\n");
        code.append("  }\n\n");
        
        return code.toString();
    }
}
```
